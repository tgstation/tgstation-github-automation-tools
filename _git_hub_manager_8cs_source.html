381  public async Task<string> CheckAuthorization(IRequestCookieCollection cookies, CancellationToken cancellationToken)
382  {
383  if (!cookies.TryGetValue(CookieName, out string cookieGuid))
384  return null;
385 
386  if (!Guid.TryParse(cookieGuid, out Guid guid))
387  return null;
388 
389  //cleanup
390  var now = DateTimeOffset.Now;
391  var everything = await databaseContext.AccessTokenEntries.ToAsyncEnumerable().ToList().ConfigureAwait(false);
392  var toRemove = everything.Where(x => x.Expiry < now);
393  databaseContext.AccessTokenEntries.RemoveRange(toRemove);
394  await databaseContext.Save(cancellationToken).ConfigureAwait(false);
395 
396  var entry = everything.Where(x => x.Id == guid && x.Expiry >= now).FirstOrDefault();
397  if (entry == default(AccessTokenEntry))
398  return null;
399 
400  return entry.AccessToken;
401  }
