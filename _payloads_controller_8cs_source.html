85  bool CheckPayloadSignature(string payload, string signatureWithPrefix)
86  {
87  const string Sha1Prefix = "sha1=";
88  if (!signatureWithPrefix.StartsWith(Sha1Prefix, StringComparison.OrdinalIgnoreCase))
89  return false;
90  var signature = signatureWithPrefix.Substring(Sha1Prefix.Length);
91  var secret = Encoding.UTF8.GetBytes(gitHubConfiguration.WebhookSecret);
92  var payloadBytes = Encoding.UTF8.GetBytes(payload);
93 
94  byte[] hash;
95 #pragma warning disable CA5350 // Do not use insecure cryptographic algorithm SHA1.
96  using (var hmSha1 = new HMACSHA1(secret))
97 #pragma warning restore CA5350 // Do not use insecure cryptographic algorithm SHA1.
98  hash = hmSha1.ComputeHash(payloadBytes);
99  var expectedHash = ToHexString(hash);
100  logger.LogTrace("Expect: {0}. Received: {1}", expectedHash, signature);
101  return expectedHash == signature;
